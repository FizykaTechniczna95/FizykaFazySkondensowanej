\subsubsection*{Rozwiązanie:}
Pole koła: $\pi k_f^2$
\\
\\
element powierzchni: $(\frac{2\pi}{L})^2$
\\
\\
\begin{equation}
N=\frac{L^2k_f^2}{4\pi}
\end{equation}
\\
\begin{equation}
f_f=\frac{4\pi N}{L^2} 
\end{equation}
\\
\begin{equation}
E_f=\frac{\hbar^2}{2m}\frac{4\pi N}{L^2}
\end{equation}
\\
\begin{equation}
N=\frac{mL^2E_f}{2\pi \hbar^2}
\end{equation}
\\
\begin{equation}
\frac{dN}{d\omega}=frac{mL^2}{2\pi \hbar^2}
\end{equation}
